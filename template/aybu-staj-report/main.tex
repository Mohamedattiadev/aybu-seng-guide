\newcommand{\SigBlock}{%

    \dotfill & \dotfill \\
    Signature of trainee & Work place \\[1.4em]

  \end{tabular}%
}

% Single weekly table. Rows are passed positionally for clarity.
\newcommand{\WeekRow}[3]{#1 & #2 & #3 \\ \hline}

\newenvironment{weekreport}[3]{%
  % #1 = week label (e.g. "Week 1")
  % #2 = start date
  % #3 = end date
  \subsection{#1}
  \noindent\textbf{Dates: #2 -- #3} \hfill \textbf{Place: \InternshipPlace}\\[0.4em]
  \renewcommand{\arraystretch}{1.6}
  \begin{tabular}{|>{\centering\arraybackslash}p{2.6cm}|p{10.2cm}|>{\centering\arraybackslash}p{2.6cm}|}
  \hline
  \textbf{DAYS} & \textbf{WORK ACCOMPLISHED} & \textbf{WORKING HOURS} \\ \hline
}{%
  \end{tabular}\\
  \SigBlock
  \newpage
}
